var doneCount = 0;
var geocoder=null, ps=null;

function render(){
  var tbody = document.getElementById('tbody');
  tbody.innerHTML = STORES_ADDR.map(function(s,i){
    var r = results[i];
    if(!r) return '<tr><td>'+s.storesName+'</td><td>'+s.addr+'</td><td>수집 중...</td></tr>';
    if(!r.ok) return '<tr><td>'+s.storesName+'</td><td>'+s.addr+'</td><td style="color:#dc2626">수집 실패</td></tr>';

function pagedKeyword(loc,keyword,maxPages,radius,onEach,onDone){
  var pc=0;
  ps.keywordSearch(keyword,function(data,status,pagination){
    if(status===kakao.maps.services.Status.OK){data.forEach(onEach);}
    pc++;
    if(status===kakao.maps.services.Status.OK&&pagination&&pagination.hasNextPage&&pc<maxPages){pagination.nextPage();}
    else{onDone();}
  },{location:loc,radius:radius,size:15});
}

function collectOne(i,cb){
  var s = STORES_ADDR[i];
  geocoder.addressSearch(s.addr, function(res,status){
    if(status!==kakao.maps.services.Status.OK || !res[0]){
      results[i]={ok:false}; doneCount++; render(); cb(); return;
    }
    var loc = new kakao.maps.LatLng(parseFloat(res[0].y), parseFloat(res[0].x));
    var intersectionDist=null;
    pagedKeyword(loc,'사거리',1,200,function(p){
      var d=Math.round(p.distance);
      if(d<=200&&(intersectionDist==null||d<intersectionDist)) intersectionDist=d;
    },function(){
      if(intersectionDist!=null){
        results[i]={ok:true,intersection:intersectionDist};
        doneCount++; render(); cb();
        return;
      }
      pagedKeyword(loc,'교차로',1,200,function(p){
        var d=Math.round(p.distance);
        if(d<=200&&(intersectionDist==null||d<intersectionDist)) intersectionDist=d;
      },function(){
        results[i]={ok:true,intersection:intersectionDist};
        doneCount++; render(); cb();
      });
    });
  });
}

function run(){
  var idx=0;
  function next(){
    if(idx>=STORES_ADDR.length)return;
    var cur=idx; idx++;
    collectOne(cur,function(){ setTimeout(next,150); });
  }
  next();
}

function initKakao(){
  geocoder = new kakao.maps.services.Geocoder();
  ps = new kakao.maps.services.Places();
  document.getElementById('status').textContent = '수집 시작...';
  render();
  run();
}
